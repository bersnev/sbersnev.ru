    USE magento2; - выбрать БД для дальнейшей работы

    SHOW TABLES; - просмотреть все таблицы

    zcat /home/belleze/backup/dev_2022-04-25_120213.sql.gz | mysql -u 'root' magento2 - восстановить БД из бэкапа

    sudo mysqldump -v -p magento2 | gzip > /home/ubuntu/magento2_27_04_22.sql.gz - создать бэкап БД
