Команды Ubuntu

Привелегии

    sudo command - запустить команду как root

    sudo -s - открыть оболочку root

    sudo -s -u user - открыть оболочку как пользователь

    sudo -k - восстановить пароль sudo

    sudo visudo - редактировать /etc/sudoers

    passwd - изменить ваш пароль

Сеть

    ifconfig - показать информацию о сети

    iwconfig - показать информацию о беспроводной сети

    sudo iwlist scan - поиск беспроводных сетей

    sudo /etc/init.d/networking restart - перезапустить сеть

    /etc/network/interfaces - файл для ручной настройки сети

    ifup interface - включить интерфейс

    ifdown interface - отключить интерфейс

    whois domain - получить информацию whois для domain

    wget file - скачать file

    ip link show - отобразить состояние всех интерфейсов

    cat /proc/net/dev - показать сетевые интерфейсы и статистику по ним

    dig domain - получить DNS информацию domain

Управление пакетами

    dpkg -l - список установленных пакетов

    apt-get update - обновить доступные обновления

    apt-get upgrade - обновить все пакеты

    apt-get install pkg - установить пакет (pkg)

    apt-get purge pkg - удалить пакет (pkg)

    apt-get autoremove - удалить устаревшие пакеты

    apt-get -f install - попробовать исправить битые пакеты

Система

    lsb_release -a - получить версию Ubuntu

    ntop - диспетчер задач

    ps -ef | grep 1477 - посмотреть процесс

    free -h - загрузка памяти

    date - отображает текущую дату и время

    uname -r - получить версию ядра

    uname -a - получить всю информацию о ядре

    arch - отобразить архитектуру компьютера

    cat /proc/cpuinfo - показать информацию о ЦПУ

    cat /proc/meminfo - проверить использование памяти

    df -h - информация об использовании дисков

    ps - процессы, в настоящее время работающие на системе

    hdparm -i /dev/sda - вывести характеристики жесткого диска

    man ls - help по работе команды ls

    sudo chmod 777 /var/www/html - дать все права на папку(755 все для владельца, остальным только чтение и выполнение)

Файловые команды

    cd /home - перейти в директорию '/home'

    cd .. - перейти в директорию уровнем выше

    cd - перейти в домашнюю директорию

    cd - - перейти в директорию, в которой находились до перехода в текущую директорию

    mkdir dir1 - создать директорию с именем 'dir1'

    rmdir dir1 - удалить директорию с именем 'dir1'

    touch help.txt - создать текстовый файл с именем 'help'

    rm -f file1 - удалить файл с именем 'file1

    grep -i serg /user/ubuntu/help.txt - найти пользователя serg в файле

    grep -i -B4 serg /user/ubuntu/help.txt - найти пользователя serg в файле и вывести строку с ним

    grep -r "serg" /user/ubuntu/ - найти serg и производные в файлах папки ubuntu и вывести строку с ним

    find . -iname "TeStFile*" - Найти в текущей папке файл с именем TeStFile

    find / -size +50M -size -100M - Найти файлы от пятидесяти до ста мегабайт

    tar -cvzf archive.tar.gz /home/test - заархивировать папку test

    tar -xvf archive.tar.gz - разархивировать файл archive.tar.gz

    mv dir1 new_dir - переименовать или переместить файл или директорию

    cp file1 file2 - сопировать файл file1 в файл file2

    cp dir/* . - копировать все файлы директории dir в текущую директорию

    cp -a /tmp/dir1 . - копировать директорию dir1 со всем содержимым в текущую директорию

    cp -a dir1 dir2 - копировать директорию dir1 в директорию dir2

Пользователи и группы

    whoami - имя, под которым вы залогинены

    groupadd group_name - создать новую группу с именем group_name

    groupdel group_name - удалить группу group_name

    useradd user1 - создать пользователя user1

    userdel -r user1 - удалить пользователя user1 и его домашний каталог

    passwd user1 - сменить пароль пользователя user1 (только root)
